bool estaOrdenado(std::vector<Celda> v){

    bool ordenado = true;
    int comprobado = 0;
    
    for(int i=0; i<v.size() && (comprobado==0); i++){
        if(v[i].valor < v[i+1].valor){
        	comprobado++;
            ordenado = false;
        }
	}
	
    return ordenado;
}
